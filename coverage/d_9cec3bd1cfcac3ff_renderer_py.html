26class Order(enum.Enum): 

27 """Enumeration for the possible members ordering.""" 

28 

29 alphabetical = "alphabetical" 

30 source = "source"

33def _sort_key_alphabetical(item: CollectorItem) -> Any: 

34 # chr(sys.maxunicode) is a string that contains the final unicode 

35 # character, so if 'name' isn't found on the object, the item will go to 

36 # the end of the list. 

37 return item.name or chr(sys.maxunicode) 

38 

39 

40def _sort_key_source(item: CollectorItem) -> Any: 

41 # if 'lineno' is none, the item will go to the start of the list. 

42 return item.lineno if item.lineno is not None else -1 

43 

44 

45order_map = { 

46 Order.alphabetical: _sort_key_alphabetical, 

47 Order.source: _sort_key_source, 

48}

109 def render(self, data: CollectorItem, config: dict) -> str: # noqa: D102 (ignore missing docstring) 

110 final_config = ChainMap(config, self.default_config) 

111 

112 template = self.env.get_template(f"{data.kind.value}.html") 

113 

114 # Heading level is a "state" variable, that will change at each step 

115 # of the rendering recursion. Therefore, it's easier to use it as a plain value 

116 # than as an item in a dictionary. 

117 heading_level = final_config["heading_level"] 

118 try: 

119 final_config["members_order"] = Order(final_config["members_order"]) 

120 except ValueError: 

121 choices = "', '".join(item.value for item in Order) 

122 raise PluginError(f"Unknown members_order '{final_config['members_order']}', choose between '{choices}'.") 

123 

124 return template.render( 

125 **{"config": final_config, data.kind.value: data, "heading_level": heading_level, "root": True}, 

126 )

141 def do_format_signature(self, signature: str, line_length: int) -> str: 

142 """Format a signature using Black. 

143 

144 Parameters: 

145 signature: The signature to format. 

146 line_length: The line length to give to Black. 

147 

148 Returns: 

149 The same code, formatted. 

150 """ 

151 code = signature.strip() 

152 if len(code) < line_length: 

153 return code 

154 formatter = _get_black_formatter() 

155 return formatter(code, line_length)

186 def do_multi_crossref(self, text: str, code: bool = True) -> Markup: 

187 """Filter to create cross-references. 

188 

189 Parameters: 

190 text: The text to scan. 

191 code: Whether to wrap the result in a code tag. 

192 

193 Returns: 

194 Markup text. 

195 """ 

196 group_number = 0 

197 variables = {} 

198 

199 def repl(match): # noqa: WPS430 

200 nonlocal group_number # noqa: WPS420 

201 group_number += 1 

202 path = match.group() 

203 path_var = f"path{group_number}" 

204 variables[path_var] = path 

205 return f"<span data-autorefs-optional-hover={{{path_var}}}>{{{path_var}}}</span>" 

206 

207 text = re.sub(r"([\w.]+)", repl, text) 

208 if code: 

209 text = f"<code>{text}</code>" 

210 return Markup(text).format(**variables)

213@lru_cache(maxsize=1) 

214def _get_black_formatter(): 

215 try: 

216 from black import Mode, format_str 

217 except ModuleNotFoundError: 

218 logger.warning("Formatting signatures requires Black to be installed.") 

219 return lambda text, _: text

221 def formatter(code, line_length): # noqa: WPS430 

222 mode = Mode(line_length=line_length) 

223 formatted = format_str(f"def {code}: pass", mode=mode) 

224 # remove starting `def ` and trailing `: pass` 

225 return formatted[4:-5].strip()[:-1]

227 return formatter
